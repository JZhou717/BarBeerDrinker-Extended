QEUERY 1

#Verification for Pattern 1:
#Pattern 1:Transactions/bills cannot be issued at times when the given bar is closed
SELECT NOT EXISTS(
select *
from issues i left join bars b on i.bar = b.name
where
(SELECT ((SELECT STRCMP(i.time, b.opening_hours)) < 0) = 1) #THIS CHECKS TO SEE IF THE BILL WAS ISSUED BEFORE THE BAR OPENED;
and
(SELECT ((STRCMP(i.time, b.closing_hours)) > 0) = 1) #THIS CHECKS TO SEE IF THE BILL WAS ISSUED AFTER THE BAR CLOSED
) as issuedOnlyWhenBarOpen;



QEUERY 2

Verification for Pattern 2:
#Pattern 2:Drinkers cannot frequent bars in different state
SELECT NOT EXISTS(
select *
from frequents f left join drinkers d on f.drinker = d.name
left join bars b on b.name = f.bar
where d.address <> b.location
) as frequentsOnlyInState;



QEUERY 3

#Verification for Pattern 3:
/*Pattern 3:For every two beers, b1 and b2, different bars may charge differently for b1 and b2 but b1
should either be less expensive than b2 in ALL bars or more expensive than b2 in ALL bars.
Cannot be the case that in one bar Corona is more expensive than Bud and in another Bud is more expensive than Corona.
But Corona may be more expensive than Bud in one bar, and have the same price as Bud in another.*/
SELECT NOT EXISTS(
select *
from sells itemA left join sells itemB on itemA.bar = itemB.bar #This outer query finds all instances where itemA is cheaper or at the same price as itemB
where itemA.item <> itemB.item
and itemA.price <= itemB.price
and (itemA.bar, itemA.item) in
(select itemA1.bar, itemA1.item #This inner query checks all instances where the same itemA is more expensive than the same itemB
from sells itemA1 left join sells itemB1 on itemA1.bar = itemB1.bar
where itemA1.item <> itemB1.item
and itemA1.price > itemB1.price
and itemA.item = itemA1.item
and itemB.item = itemB1.item
and itemA.bar <> itemA1.bar)
) as soldCheaperEverywhere;
